fail2ban installation setup :
terminal : bash
sudo apt update
sudo apt install fail2ban

fail2ban config setup :
terminal : bash
sudo systemctl enable fail2ban
sudo systemctl start fail2ban

cd /etc/fail2ban

sudo nano jail.local
^ config:
  [DEFAULT]
  bantime = [time]h
  maxretry = 3
  findtime = [time]m
  banaction = ufw / nftables-multiport / other // It depends on which one you use for firewall services.
  ignoreip = [loopback]/8 ::1 [local network ip]/24 // depending on the permitted IP access
  backend = systemd

cat jail.local

sudo systemctl restart fail2ban
sudo systemctl status fail2ban

sudo journalctl -u fail2ban -f

sudo tail -f /var/log/fail2ban.log
